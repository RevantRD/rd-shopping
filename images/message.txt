        let cart=[];

        function addToCart(Pink_Dress,$20){
            const item={
                name: Pink_Dress,
                price:$20,
            };
            cart.push(item);
            updateCartUI();
        }
        function updateCartUI(){
            const cartList = document.getElementById('cart');
            cartList.inner.html='';

            cart.forEach(item =>{
                const listItem= document.createElement('li');
                listItem.textContent='${Pink.Dress}-$$($20.99.toFixed(2))';
                cartList.appendChild(listItem);
            });
        }
